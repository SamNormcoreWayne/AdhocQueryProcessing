SELECT ATTRIBUTE:
cust, prod, month, 1_sum_quant, 2_sum_quant, 3_sum_quant
NUMBER OF GROUPING VARIABLES:
3
GROUPING ATTRIBUTES:
cust, prod, month
F-VECT:
1_sum_quant, 1_avg_quant, 2_sum_quant, 3_sum_quant, 3_avg_quant
SELECT CONDITION-VECT:
1.state = NY and 1.quant < sum(quant)
2.state = NJ
3.state = CT
HAVING_CONDITION:
1_sum_quant > 2 * 2_sum_quant or 1_avg_quant > 3_avg_quant;
